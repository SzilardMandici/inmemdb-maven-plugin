17  package com.btmatthews.maven.plugins.inmemdb;
18  
19  import com.btmatthews.maven.plugins.inmemdb.db.derby.DerbyDatabase;
20  import com.btmatthews.maven.plugins.inmemdb.db.hsqldb.HSQLDBDatabase;
21  
22  /**
23   * The factory that creates database objects.
24   * 
25   * @author <a href="mailto:brian@btmatthews.com">Brian Matthews</a>
26   * @version 1.0.0
27   */
28  public class DatabaseFactory {
29  
30  	/**
31  	 * The database type code for HSQLDB databases.
32  	 */
33  	public static final String TYPE_HSQLDB = "hsqldb";
34  
35  	/**
36  	 * The database type code for Apache Derby databases.
37  	 */
38  	public static final String TYPE_DERBY = "derby";
39  
40  	/**
41  	 * Create and return a database object corresponding to type and initialise
42  	 * it with the database name, user name and password. If the type is not
43  	 * supported then <code>null</code> will be returned.
44  	 * 
45  	 * @param type
46  	 *            The database type.
47  	 * @param databaseName
48  	 *            The database name.
49  	 * @param username
50  	 *            The user name used to connect to the database.
51  	 * @param password
52  	 *            The password used to connect to the database.
53  	 * @return The database object or <code>null</code> if the database type is
54  	 *         not supported.
55  	 */
56  	public Database createDatabase(String type, String databaseName,
57  			String username, String password) {
58  		Database database;
59  		if (TYPE_HSQLDB.equals(type)) {
60  			database = new HSQLDBDatabase(databaseName, username, password);
61  		} else if (TYPE_DERBY.equals(type)) {
62  				database = new DerbyDatabase(databaseName, username, password);
63  		} else {
64  			database = null;
65  		}
66  		return database;
67  	}
68  }

27  /**
28   * Abstract base classes for database objects providing an implentation of the
29   * {@link load()} operation that is used to load data into or execute scripts
30   * against the database.
31   * 
32   * @author <a href="mailto:brian@btmatthews.com">Brian Matthews</a>
33   * @version 1.0.0
34   */
35  public abstract class AbstractDatabase implements Database {
36  
37  	/**
38  	 * The message key for the error reported when a file type is not supported.
39  	 */
40  	private static final String UNSUPPORTED_FILE_TYPE = "unsupported_file_type";
41  
42  	/**
43  	 * The message key for the error reported when a server cannot be started.
44  	 */
45  	protected static final String ERROR_STARTING_SERVER = "error_starting_server";
46  
47  	/**
48  	 * The message key for the error reported when a server cannot be stopped.
49  	 */
50  	protected static final String ERROR_STOPPING_SERVER = "error_stopping_server";
51  
52  	/**
53  	 * The database name.
54  	 */
55  	private String databaseName;
56  
57  	/**
58  	 * The user name used to connect to the database.
59  	 */
60  	private String username;
61  
62  	/**
63  	 * The password used to connect to the database.
64  	 */
65  	private String password;
66  
67  	/**
68  	 * The constructor for this object stores the database name, user name and
69  	 * password that will be used to create connections.
70  	 * 
71  	 * @param database
72  	 *            The database name.
73  	 * @param username
74  	 *            The user name for the database connection.
75  	 * @param password
76  	 *            The password for the database connection.
77  	 */
78  	protected AbstractDatabase(final String database, final String username,
79  			final String password) {
80  		this.databaseName = database;
81  		this.username = username;
82  		this.password = password;
83  	}
84  
85  	/**
86  	 * Get the database name.
87  	 * 
88  	 * @return The database name.
89  	 */
90  	protected final String getDatabaseName() {
91  		return this.databaseName;
92  	}
93  
94  	/**
95  	 * Get the user name used to connect to the database.
96  	 * 
97  	 * @return The user name.
98  	 */
99  	protected final String getUsername() {
100 		return username;
101 	}
102 
103 	/**
104 	 * Get the password used to connect to the database.
105 	 * 
106 	 * @return The password.
107 	 */
108 	protected final String getPassword() {
109 		return password;
110 	}
111 
112 	/**
113 	 * Get the loaders that are supported for loading data or executing scripts.
114 	 * 
115 	 * @return Returns an array of loaders.
116 	 */
117 	protected abstract Loader[] getLoaders();
118 
119 	/**
120 	 * Get a data source object without additional connection attributes.
121 	 * 
122 	 * @return The data source object.
123 	 */
124 	public final DataSource getDataSource() {
125 		final Map<String, String> attributes = new HashMap<String, String>();
126 		return getDataSource(attributes);
127 	}
128 
129 	/**
130 	 * Find the loader that supports the source file and use it to load the data
131 	 * into or execute the script against the database.
132 	 * 
133 	 * @param logger
134 	 *            Used to report errors and raise exceptions.
135 	 * @param source
136 	 *            The source file containing data or script.
137 	 * @throws MojoFailureException
138 	 *             If there was an error loading the data or executing the
139 	 *             script or if the source file type was not supported.
140 	 */
141 	public final void load(final Logger logger, final File source)
142 			throws MojoFailureException {
143 		if (source == null) {
144 			logger.logError(UNSUPPORTED_FILE_TYPE, "null");
145 		} else {
146 			final Loader[] loaders = getLoaders();
147 			int i = 0;
148 			while (i < loaders.length) {
149 				if (loaders[i].isSupported(source)) {
150 					loaders[i].load(logger, this, source);
151 					break;
152 				}
153 				++i;
154 			}
155 			if (i >= loaders.length) {
156 				logger.logError(UNSUPPORTED_FILE_TYPE, source.getPath());
157 			}
158 		}
159 	}
160 }
